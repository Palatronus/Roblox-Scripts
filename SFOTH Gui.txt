local Players = game:GetService("Players")
local UIS = game:GetService("UserInputService")
local LP = Players.LocalPlayer
local Toggle = true

-- Universal toggle system
UIS.InputBegan:Connect(function(Input)
    if Input.KeyCode == Enum.KeyCode.H then
        Toggle = not Toggle
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "SYSTEM: "..(Toggle and "ON" or "OFF"),
            Duration = 1
        })
    end
end)

local function MassTrigger(models)
    if not LP.Character then return end
    local root = LP.Character:FindFirstChild("HumanoidRootPart")
    if not root then return end

    for _, model in ipairs(models) do
        local part = model:FindFirstChildWhichIsA("Part")
        if part then
            -- Double-tap interaction pattern
            firetouchinterest(root, part, 0)
            task.wait(0.05)
            firetouchinterest(root, part, 1)
        end
    end
end

local function MaintainArmor()
    local armorStands = {}
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj.Name == "BattleArmorStand" and obj:IsA("Model") then
            table.insert(armorStands, obj)
        end
    end
    MassTrigger(armorStands)
end

local function MassHeal()
    local healPads = {}
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj.Name == "HealPad" and obj:IsA("Model") then
            table.insert(healPads, obj)
        end
    end
    MassTrigger(healPads)
end

-- Core systems
task.spawn(function()
    while task.wait(0.5) do
        if Toggle and LP.Character then
            -- Armor maintenance
            if not LP.Character:FindFirstChild("BattleArmor") then
                MaintainArmor()
            end
            
            -- Health maintenance
            if LP.Character:FindFirstChild("Humanoid") and 
               LP.Character.Humanoid.Health < LP.Character.Humanoid.MaxHealth then
                MassHeal()
            end
        end
    end
end)

-- Initial notification
game:GetService("StarterGui"):SetCore("SendNotification", {
    Title = "SFOTH ULTIMATE SCRIPT",
    Text = "Brute-force armor/heal\nAll items utilized",
    Duration = 5
})
