-- Roblox Executor Script for Sword Fight on the Heights Original

-- Variables for armor stands, heal pads, and your player character
local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:WaitForChild("Humanoid")
local workspace = game:GetService("Workspace")

-- Armor stand and heal pad names
local armorStandName = "BattleArmorStand"
local healPadName = "HealPad"

-- Function to check if a heal pad is active (green)
local function isHealPadActive(healPad)
    return healPad.BrickColor == BrickColor.new("Bright green")
end

-- Function to check if armor is equipped
local function equipArmor()
    local armorStand = workspace:FindFirstChild(armorStandName)
    if armorStand and armorStand:IsA("Model") then
        -- Equip armor if it is available and not taken
        if not armorStand:FindFirstChild("Equipped") then
            -- Set the armor as equipped
            local equipped = Instance.new("BoolValue")
            equipped.Name = "Equipped"
            equipped.Parent = armorStand
            -- Trigger armor equip action here (adjust according to how the armor works)
            -- Example: Equip some gear here or give health boosts
            humanoid.MaxHealth = humanoid.MaxHealth * 2 -- Double health on armor equip
            humanoid.Health = humanoid.MaxHealth -- Restore health after equip
        end
    end
end

-- Function to trigger healing (heal pad interaction)
local function triggerHealPad(healPad)
    if isHealPadActive(healPad) then
        -- Activate healing if heal pad is green
        humanoid.Health = humanoid.MaxHealth -- Heal player completely
    end
end

-- Function to monitor health and trigger actions
local function monitorPlayerHealth()
    -- Heal player if health is low and a heal pad is available
    while true do
        wait(0.5) -- Loop every 0.5 seconds to check and perform actions without freezing the game

        -- Check if health is low and use heal pad
        local healPad = workspace:FindFirstChild(healPadName)
        if healPad then
            if isHealPadActive(healPad) then
                triggerHealPad(healPad) -- Trigger heal pad if it's active
            end
        end

        -- Equip armor if possible
        equipArmor()

        -- Check if player has taken damage and heal immediately
        if humanoid.Health < humanoid.MaxHealth then
            -- If health is below max, heal
            local healPads = workspace:GetChildren()
            for _, pad in pairs(healPads) do
                if pad.Name == healPadName and isHealPadActive(pad) then
                    triggerHealPad(pad)
                    break
                end
            end
        end
    end
end

-- Function to ensure the script continues running
local function keepScriptActive()
    while true do
        wait(5) -- Check every 5 seconds to ensure everything keeps working
        -- Recheck armor and healing systems to ensure script keeps running properly
        equipArmor()
        monitorPlayerHealth()
    end
end

-- Start the script by monitoring the player's health and armor
monitorPlayerHealth()
keepScriptActive()
