local player = game.Players.LocalPlayer
local workspace = game.Workspace

function checkArmorStand()
    for _, armorStand in pairs(workspace:GetChildren()) do
        if armorStand.Name == "BattleArmorStand" and not armorStand:FindFirstChild("Touched") then
            player.Character.Humanoid:EquipTool(armorStand)
        end
    end
end

function checkHealPad()
    for _, healPad in pairs(workspace:GetChildren()) do
        if healPad.Name == "HealPad" then
            if healPad.BrickColor.Name == "Bright green" then
                -- Heal pad is active, do nothing
            elseif healPad.BrickColor.Name == "Bright red" then
                healPad.Touched:Fire()
            end
        end
    end
end

function mainLoop()
    while true do
        checkArmorStand()
        checkHealPad()
        wait(0.5) -- Adjust this timing to balance performance and responsiveness
    end
end

mainLoop()
