-- Client-Sided Reanimation Script with "Perma Death"
-- Use responsibly and only for learning or private purposes.

local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:WaitForChild("Humanoid")
local camera = workspace.CurrentCamera

-- Settings
local permaDeathPartName = "Left Arm" -- Name of the part to "perma-death"
local reanimationDelay = 0.1 -- Delay before reanimation starts

-- Function to remove default connections and control parts manually
local function reanimate()
    -- Wait for the reanimation delay
    wait(reanimationDelay)

    -- Clone the character
    local clone = character:Clone()
    clone.Parent = workspace
    clone.Name = "Reanimated_" .. player.Name

    -- Find the limb for "Perma Death" and anchor it
    for _, part in pairs(character:GetDescendants()) do
        if part:IsA("BasePart") and part.Name == permaDeathPartName then
            part.Anchored = true
            part.BrickColor = BrickColor.new("Really red") -- Visual cue
        end
    end

    -- Set transparency of the original character to make it "dead"
    for _, part in pairs(character:GetDescendants()) do
        if part:IsA("BasePart") and part.Name ~= permaDeathPartName then
            part.Transparency = 1
        end
    end

    -- Transfer camera to the clone
    local rootPart = clone:FindFirstChild("HumanoidRootPart")
    if rootPart then
        camera.CameraSubject = clone:FindFirstChild("Humanoid")
        rootPart.CFrame = character:FindFirstChild("HumanoidRootPart").CFrame
