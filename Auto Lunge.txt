-- Auto Sword Lunging Script (FE Compatible)
local Player = game:GetService("Players").LocalPlayer
local UIS = game:GetService("UserInputService")
local RS = game:GetService("RunService")

local ToolName = "Sword"
local Toggle = false
local Cooldown = 0.01 -- Adjust cooldown between lunges
local Debounce = false

local function GetEquippedSword()
    if Player.Character then
        for _, tool in ipairs(Player.Character:GetChildren()) do
            if tool:IsA("Tool") and tool.Name == ToolName then
                return tool
            end
        end
    end
    return nil
end

UIS.InputBegan:Connect(function(input, processed)
    if not processed and input.KeyCode == Enum.KeyCode.F then
        if Debounce then return end
        Debounce = true
        
        Toggle = not Toggle
        print("Auto Lunge:", Toggle and "Enabled" or "Disabled")
        
        if Toggle then
            coroutine.wrap(function()
                while Toggle and wait(Cooldown) do
                    local sword = GetEquippedSword()
                    if sword then
                        sword:Activate()
                    end
                end
            end)()
        end
        
        Debounce = false
    end
end)

-- Visual feedback
RS.Heartbeat:Connect(function()
    if Toggle then
        -- Optional: Add visual effects here
    end
end)

print("Auto Lunge Script Loaded | Press F to Toggle")
