local Players = game:GetService("Players")

local VOID_COORDINATES = CFrame.new(0, -1000000, 0)

local function teleportToVoid(character)
    if character then
        character:PivotTo(VOID_COORDINATES)
    end
end

local function onPlayerAdded(player)
    player.CharacterAdded:Connect(teleportToVoid)

    if player.Character then
        teleportToVoid(player.Character)
    end
end

for _, player in ipairs(Players:GetPlayers()) do
    onPlayerAdded(player)
end

Players.PlayerAdded:Connect(onPlayerAdded)
