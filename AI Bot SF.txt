-- Simple test script to verify executor functionality
print("TestScript_Simple: Starting...")

-- Create a basic GUI to confirm script execution
local function CreateTestGUI()
    local ScreenGui = Instance.new("ScreenGui")
    ScreenGui.Name = "TestScript_Status"
    
    -- Try different parent options
    local success = pcall(function()
        ScreenGui.Parent = game:GetService("CoreGui")
    end)
    
    if not success then
        success = pcall(function()
            ScreenGui.Parent = game:GetService("Players").LocalPlayer.PlayerGui
        end)
    end
    
    local Frame = Instance.new("Frame")
    Frame.Size = UDim2.new(0, 200, 0, 100)
    Frame.Position = UDim2.new(0.5, -100, 0.5, -50)
    Frame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    Frame.BackgroundTransparency = 0.5
    Frame.BorderSizePixel = 2
    Frame.Parent = ScreenGui
    
    local Title = Instance.new("TextLabel")
    Title.Size = UDim2.new(1, 0, 0, 20)
    Title.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    Title.TextColor3 = Color3.fromRGB(255, 255, 255)
    Title.Text = "Test Script Running"
    Title.Font = Enum.Font.SourceSansBold
    Title.TextSize = 14
    Title.Parent = Frame
    
    local Status = Instance.new("TextLabel")
    Status.Size = UDim2.new(1, 0, 1, -20)
    Status.Position = UDim2.new(0, 0, 0, 20)
    Status.BackgroundTransparency = 1
    Status.TextColor3 = Color3.fromRGB(255, 255, 255)
    Status.Text = "Script is active"
    Status.Font = Enum.Font.SourceSans
    Status.TextSize = 14
    Status.TextXAlignment = Enum.TextXAlignment.Left
    Status.TextYAlignment = Enum.TextYAlignment.Top
    Status.Parent = Frame
    
    return Status
end

-- Main execution
local success, result = pcall(function()
    print("TestScript_Simple: Creating GUI...")
    local statusLabel = CreateTestGUI()
    
    -- Update status periodically to confirm script is still running
    local counter = 0
    
    while true do
        counter = counter + 1
        if statusLabel then
            statusLabel.Text = "Script is active\nCounter: " .. counter
        end
        print("TestScript_Simple: Counter = " .. counter)
        wait(1)
    end
end)

if not success then
    print("TestScript_Simple ERROR: " .. tostring(result))
end
