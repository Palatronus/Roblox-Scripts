-- Direct HRP Welding Manipulation
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer

local function breakHRPWelds()
    local character = LocalPlayer.Character
    if not character then return false end
    
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    
    -- Break welds connecting HRP to other parts
    for _, weld in ipairs(hrp:GetChildren()) do
        if weld:IsA("Weld") or weld:IsA("WeldConstraint") then
            weld.Enabled = false
        end
    end
    
    -- Also break welds from torso to HRP
    local torso = character:FindFirstChild("UpperTorso") or character:FindFirstChild("Torso")
    if torso then
        for _, weld in ipairs(torso:GetChildren()) do
            if (weld:IsA("Weld") or weld:IsA("WeldConstraint")) and weld.Part1 == hrp then
                weld.Enabled = false
            end
        end
    end
    
    return true
end

local function separateHRP()
    if not breakHRPWelds() then
        warn("Failed to break HRP welds")
        return
    end
    
    local character = LocalPlayer.Character
    local hrp = character:FindFirstChild("HumanoidRootPart")
    
    hrp.Anchored = true
    hrp.CFrame = CFrame.new(100, 50, 200) -- Your target position
    
    print("HRP welds broken - HRP separated from character")
end

-- Execute immediately
separateHRP()
