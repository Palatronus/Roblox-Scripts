-- Target coordinates (X, Y, Z)
local targetPosition = Vector3.new(100, 50, 200)
-- Control variable for the locking loop
local keepTeleporting = true

local function teleportLoop()
    -- Use pcall for error handling :cite[5]
    local success, errorMessage = pcall(function()
        local player = game:GetService("Players").LocalPlayer
        -- Wait for character to load
        if not player.Character then
            player.CharacterAdded:Wait()
        end
        local character = player.Character
        -- Find the HumanoidRootPart with error checking :cite[7]
        local humanoidRootPart = character:WaitForChild("HumanoidRootPart")
        
        while keepTeleporting and task.wait(0.1) do -- Loop to maintain position
            if humanoidRootPart and humanoidRootPart.Parent then
                -- Set the CFrame to the target position :cite[3]:cite[7]
                humanoidRootPart.CFrame = CFrame.new(targetPosition)
            else
                break -- Exit loop if HRP is no longer valid
            end
        end
    end)
    
    if not success then
        warn("An error occurred: " .. errorMessage)
    end
end

-- Start the teleportation loop
teleportLoop()

-- To stop the teleportation, set 'keepTeleporting' to false
