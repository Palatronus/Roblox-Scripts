-- Simple HRP Anchor Script
local Players = game:GetService("Players")
local player = Players.LocalPlayer

local function AnchorHRP()
    if player.Character then
        local hrp = player.Character:FindFirstChild("HumanoidRootPart")
        if hrp then
            hrp.Anchored = true
            hrp.CanCollide = false
            hrp.Transparency = 1 -- Make invisible
            hrp.CFrame = CFrame.new(0, 1000, 0) -- Move to safe location
            
            -- Create visible decoy at original position
            local decoy = hrp:Clone()
            decoy.Name = "VisibleDecoy"
            decoy.Transparency = 0
            decoy.Anchored = true
            decoy.CFrame = hrp.CFrame
            decoy.Parent = player.Character
        end
    end
end

-- Execute immediately
AnchorHRP()

-- Re-anchor on respawn
player.CharacterAdded:Connect(AnchorHRP)
