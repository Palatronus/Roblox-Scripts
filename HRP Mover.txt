-- HRP Teleport Script for Executors
-- Paste this entire script into your executor

if not game:IsLoaded() then
    game.Loaded:Wait()
end

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer

-- Wait for character to load
if not LocalPlayer.Character then
    LocalPlayer.CharacterAdded:Wait()
end

local Character = LocalPlayer.Character
local HumanoidRootPart = Character:WaitForChild("HumanoidRootPart")

-- Target coordinates (modify these as needed)
local TargetPosition = Vector3.new(100, 50, 200) -- X, Y, Z coordinates
local TeleportActive = true

local function TeleportHRP()
    local success, err = pcall(function()
        -- Main teleportation loop
        while TeleportActive do
            if Character and Character.Parent and HumanoidRootPart and HumanoidRootPart.Parent then
                -- Direct CFrame manipulation for teleportation
                HumanoidRootPart.CFrame = CFrame.new(TargetPosition)
                
                -- Force position maintenance
                HumanoidRootPart.Velocity = Vector3.new(0, 0, 0)
                HumanoidRootPart.RotVelocity = Vector3.new(0, 0, 0)
                
                -- Small delay to prevent crashing
                wait(0.1)
            else
                -- Re-acquire character if respawned
                Character = LocalPlayer.Character
                if Character then
                    HumanoidRootPart = Character:WaitForChild("HumanoidRootPart")
                end
                wait(1)
            end
        end
    end)
    
    if not success then
        warn("Teleport Error: " .. tostring(err))
    end
end

-- Start teleportation
TeleportHRP()

-- Optional: Keybind to toggle teleport
local UIS = game:GetService("UserInputService")
UIS.InputBegan:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.P then  -- Press P to toggle
        TeleportActive = not TeleportActive
        print("Teleport " .. (TeleportActive and "enabled" or "disabled"))
    end
end)

print("HRP Teleport Script Activated")
print("Target Position: " .. tostring(TargetPosition))
print("Press P to toggle teleportation")
