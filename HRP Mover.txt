-- Intermediate HRP Test
print("=== HRP FUNCTIONALITY TEST ===")

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

-- Wait for everything to load
if not game:IsLoaded() then
    print("Waiting for game to load...")
    game.Loaded:Wait()
end

local LocalPlayer = Players.LocalPlayer
print("Player: " .. LocalPlayer.Name)

-- Wait for character
if not LocalPlayer.Character then
    print("Waiting for character...")
    LocalPlayer.CharacterAdded:Wait()
end

local character = LocalPlayer.Character
print("Character: " .. character.Name)

-- Find HRP
local hrp = character:WaitForChild("HumanoidRootPart")
print("HRP found at: " .. tostring(hrp.Position))

-- Simple teleport test
print("Testing teleport...")
local originalPos = hrp.Position
hrp.CFrame = CFrame.new(originalPos + Vector3.new(0, 10, 0))
print("Teleported to: " .. tostring(hrp.Position))
wait(1)
hrp.CFrame = CFrame.new(originalPos)
print("Returned to original position")

print("=== TEST COMPLETE ===")
