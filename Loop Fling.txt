-- Nameless Admin-Style Prediction Fling
local Flinging = false
local function FlingTarget(target)
    while Flinging and task.wait() do
        if target.Character and LP.Character then
            local tHRP = target.Character:FindFirstChild("HumanoidRootPart")
            local myHRP = LP.Character:FindFirstChild("HumanoidRootPart")
            
            if tHRP and myHRP then
                -- Classic Nameless Admin pattern
                local pattern = {
                    CFrame.new(5, 3, 5),
                    CFrame.new(-5, 3, -5),
                    CFrame.new(5, 3, -5),
                    CFrame.new(-5, 3, 5),
                    CFrame.new(8, 5, 0),
                    CFrame.new(-8, 5, 0)
                }

                -- Network control
                if tHRP:GetNetworkOwner() ~= LP then
                    tHRP:SetNetworkOwner(LP)
                end

                -- Prediction cycle
                for _, cf in ipairs(pattern) do
                    if not Flinging then break end
                    
                    -- Position and velocity
                    myHRP.CFrame = tHRP.CFrame * cf
                    myHRP.Velocity = (tHRP.Position - myHRP.Position).Unit * 7500
                                  + Vector3.new(0, 4500, 0)
                    
                    -- Force transfer
                    tHRP.Velocity = (myHRP.Position - tHRP.Position).Unit * 6000
                                   + Vector3.new(0, 3500, 0)
                    
                    -- Collision control
                    myHRP.CanCollide = false
                    tHRP.CanCollide = false
                    
                    task.wait(0.03)
                end
            end
        end
    end
end
