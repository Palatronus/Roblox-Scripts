-- Modified Fling System (Only this block changes)
local Flinging = false
local function FlingTarget(target)
    while Flinging and task.wait(0.01) do
        if target.Character and LP.Character then
            local tHRP = target.Character:FindFirstChild("HumanoidRootPart")
            local myHRP = LP.Character:FindFirstChild("HumanoidRootPart")
            
            if tHRP and myHRP then
                -- Chaotic Orbital Pattern
                local angle = tick() * 25 -- Faster rotation
                local offset = Vector3.new(
                    math.cos(angle) * 7, 
                    math.sin(angle * 2) * 3 + 5, 
                    math.sin(angle) * 7
                )
                
                -- Nuclear Forces
                myHRP.CFrame = tHRP.CFrame + offset
                myHRP.Velocity = (tHRP.Position - myHRP.Position).Unit * 18000 + Vector3.new(0, 9000, 0)
                
                -- Target Disruption
                tHRP.Velocity = myHRP.Velocity * 0.6
                myHRP.RotVelocity = Vector3.new(math.random(-2000,2000), math.random(-2000,2000), math.random(-2000,2000))
                
                -- Collision Control
                myHRP.CanCollide = false
                tHRP.CanCollide = false
            end
        end
    end
end
