cmd.add({"fling3"}, {"fling3 <player>", "Prediction-based fling"}, function(...)
    local Target = getPlr(...)
    if not Target then return end
    
    local Character = Target.Character
    if not Character then return end
    
    local RootPart = Character:FindFirstChild("HumanoidRootPart")
    if not RootPart then return end

    -- Calculate fling direction (player's velocity + upward force)
    local flingPower = 5000
    local flingDirection = (RootPart.Velocity * 2) + Vector3.new(0, flingPower/2, 0)
    
    -- Apply prediction fling
    local BV = Instance.new("BodyVelocity")
    BV.Velocity = flingDirection
    BV.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    BV.P = 1250
    BV.Parent = RootPart
    
    -- Cleanup after 0.1 seconds
    game:GetService("Debris"):AddItem(BV, 0.1)
    
    Notify({
        Description = "Flinging "..Target.Name.."!",
        Title = "Nameless Admin",
        Duration = 3
    })
end)
