-- Fling System Core
local Fling = {
    Active = false,
    Original = {
        CFrame = nil,
        Velocity = nil,
        RotVelocity = nil
    },
    Connections = {},
    Anchors = {},
    Target = nil
}

local function CleanAll()
    -- Restore original state
    if Fling.Original.CFrame and LocalPlayer.Character then
        local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if hrp then
            hrp.CFrame = Fling.Original.CFrame
            hrp.Velocity = Fling.Original.Velocity
            hrp.RotVelocity = Fling.Original.RotVelocity
        end
    end

    -- Cleanup connections and parts
    for _, c in pairs(Fling.Connections) do c:Disconnect() end
    for _, a in pairs(Fling.Anchors) do a:Destroy() end
    Fling.Connections = {}
    Fling.Anchors = {}
    Fling.Target = nil
end

local function TrackTarget(targetHRP)
    if not Fling.Active then return end
    
    -- Store original position
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if hrp then
        Fling.Original.CFrame = hrp.CFrame
        Fling.Original.Velocity = hrp.Velocity
        Fling.Original.RotVelocity = hrp.RotVelocity
    end

    -- Create tracking anchor
    local anchor = Instance.new("Part")
    anchor.Name = "FlingAnchor"
    anchor.Anchored = true
    anchor.Transparency = 1
    anchor.Size = Vector3.new(2, 2, 1)
    anchor.CFrame = hrp.CFrame
    anchor.Parent = LocalPlayer.Character
    table.insert(Fling.Anchors, anchor)

    -- Align parts
    local parts = {}
    for _, part in ipairs(LocalPlayer.Character:GetDescendants()) do
        if part:IsA("BasePart") then
            if part.CanCollide then
                table.insert(parts, part)
                local att0 = Instance.new("Attachment")
                local att1 = att0:Clone()
                Instance.new("AlignPosition", att0).Attachment1 = att1
                att0.Parent = part
                att1.Parent = anchor
            else
                part.Velocity = Vector3.new(0, -1000, 0)
            end
        end
    end

    -- Continuous position updates
    Fling.Connections.Track = RunService.Heartbeat:Connect(function()
        if targetHRP and targetHRP.Parent then
            anchor.CFrame = targetHRP.CFrame
        else
            -- Initiate recovery if target HRP disappears
            CleanAll()
            if Fling.Active then
                task.wait(0.5)
                InitializeFling()
            end
        end
    end)

    -- Spin mechanism
    Fling.Connections.Spin = RunService.Heartbeat:Connect(function()
        for _, part in ipairs(parts) do
            part.Velocity = Vector3.new(0, -25.05, 0)
            part.RotVelocity = Vector3.new(9999, 9999, 9999)
        end
    end)
end

local function InitializeFling()
    CleanAll()
    
    -- Get current target
    local target = Players:FindFirstChild(inputBox.Text)
    if not target then return end

    -- Wait for target character
    local char = target.Character or target.CharacterAdded:Wait()
    local hrp = char:WaitForChild("HumanoidRootPart", 2)
    if not hrp then return end

    Fling.Target = target
    TrackTarget(hrp)

    -- Setup character tracking
    Fling.Connections.CharAdded = target.CharacterAdded:Connect(function(newChar)
        local newHRP = newChar:WaitForChild("HumanoidRootPart", 2)
        if newHRP then
            CleanAll()
            TrackTarget(newHRP)
        end
    end)
end

-- Control System
toggleBtn.MouseButton1Click:Connect(function()
    Fling.Active = not Fling.Active
    toggleBtn.Text = Fling.Active and "DISABLE FLING" or "ENABLE FLING"

    if Fling.Active then
        InitializeFling()
    else
        CleanAll()
    end
end)

-- Auto-reinit on local death
LocalPlayer.CharacterAdded:Connect(function()
    if Fling.Active then
        task.wait(1)
        InitializeFling()
    end
end)
