local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local LocalPlayer = Players.LocalPlayer

-- GUI Creation
local gui = Instance.new("ScreenGui")
gui.Name = "FlingGUI"
gui.ResetOnSpawn = false

local dragFrame = Instance.new("Frame", gui)
dragFrame.Size = UDim2.new(0, 200, 0, 25)
dragFrame.Position = UDim2.new(0.5, -100, 0, 10)
dragFrame.BackgroundColor3 = Color3.new(0.2, 0.2, 0.2)
dragFrame.Active = true
dragFrame.Draggable = true

local title = Instance.new("TextLabel", dragFrame)
title.Size = UDim2.new(1, 0, 1, 0)
title.Text = "Fling Controller (Drag Me)"
title.BackgroundTransparency = 1
title.TextColor3 = Color3.new(1, 1, 1)

local inputFrame = Instance.new("Frame", gui)
inputFrame.Size = UDim2.new(0, 200, 0, 80)
inputFrame.Position = UDim2.new(0.5, -100, 0, 40)
inputFrame.BackgroundColor3 = Color3.new(0.25, 0.25, 0.25)

local textBox = Instance.new("TextBox", inputFrame)
textBox.Size = UDim2.new(0.9, 0, 0, 25)
textBox.Position = UDim2.new(0.05, 0, 0.1, 0)
textBox.PlaceholderText = "Target Username"

local toggleButton = Instance.new("TextButton", inputFrame)
toggleButton.Size = UDim2.new(0.9, 0, 0, 25)
toggleButton.Position = UDim2.new(0.05, 0, 0.6, 0)
toggleButton.Text = "Start Flinging"

gui.Parent = LocalPlayer:WaitForChild("PlayerGui")

-- Fling Logic
local isFlinging = false
local connections = {}
local anchor, originalCFrame, originalVel, originalRot

local function align(Part0, Part1)
    local att1 = Instance.new("Attachment")
    att1.Parent = Part1
    local att0 = att1:Clone()
    att0.Parent = Part0

    local ap = Instance.new("AlignPosition")
    ap.Attachment0 = att0
    ap.Attachment1 = att1
    ap.ApplyAtCenterOfMass = true
    ap.MaxForce = 9e9
    ap.MaxVelocity = 9e9
    ap.ReactionForceEnabled = false
    ap.Responsiveness = 200
    ap.RigidityEnabled = false
    ap.Parent = att0
end

local function cleanFling()
    if anchor then anchor:Destroy() end
    originalCFrame, originalVel, originalRot = nil, nil, nil
    for _, v in pairs(connections) do
        v:Disconnect()
    end
    connections = {}
end

local function resetCharacter()
    local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if hrp and originalCFrame then
        hrp.CFrame = originalCFrame
        hrp.Velocity = originalVel
        hrp.RotVelocity = originalRot
    end
end

local function startFling(targetName)
    cleanFling()
    local targetPlayer = Players:FindFirstChild(targetName)
    if not targetPlayer then return end

    local character = LocalPlayer.Character
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    originalCFrame = hrp.CFrame
    originalVel = hrp.Velocity
    originalRot = hrp.RotVelocity

    anchor = Instance.new("Part")
    anchor.Name = "FlingAnchor"
    anchor.Anchored = true
    anchor.Transparency = 1
    anchor.CanCollide = false
    anchor.Size = Vector3.new(2, 2, 1)
    anchor.Parent = character

    local parts = {}
    for _, part in ipairs(character:GetDescendants()) do
        if part:IsA("BasePart") then
            if part.CanCollide then
                table.insert(parts, part)
            else
                part.Velocity = Vector3.new(0, -1000, 0)
            end
        end
    end

    for _, part in ipairs(parts) do
        align(part, anchor)
    end

    table.insert(connections, RunService.Stepped:Connect(function()
        local targetChar = targetPlayer.Character
        local targetHrp = targetChar and targetChar:FindFirstChild("HumanoidRootPart")
        if targetHrp then
            anchor.CFrame = targetHrp.CFrame
        end
    end))

    table.insert(connections, RunService.Heartbeat:Connect(function()
        for _, part in ipairs(parts) do
            part.Velocity = Vector3.new(0, -25.05, 0)
            part.RotVelocity = Vector3.new(9999, 9999, 9999)
        end
    end))
end

toggleButton.MouseButton1Click:Connect(function()
    isFlinging = not isFlinging
    toggleButton.Text = isFlinging and "Stop Flinging" or "Start Flinging"
    
    if isFlinging then
        startFling(textBox.Text)
    else
        cleanFling()
        resetCharacter()
    end
end)

-- Handle respawns
LocalPlayer.CharacterAdded:Connect(function()
    if isFlinging then
        task.wait(1)
        startFling(textBox.Text)
    end
end)
