-- Dragging System (Working)
local dragging = false
local dragStart, startPos

Main.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = Main.Position
    end
end)

Main.InputEnded:Connect(function(input)
    dragging = false
end)

UIS.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        Main.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)

-- Suggestions Populate Function
local function UpdateSuggestions()
    Suggestions:ClearAllChildren()
    local query = Input.Text:lower()
    
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LP and player.Name:lower():find(query, 1, true) then
            local btn = Instance.new("TextButton")
            btn.Text = player.Name
            btn.Size = UDim2.new(1, -10, 0, 20)
            btn.Position = UDim2.new(0, 5, 0, #Suggestions:GetChildren() * 22)
            btn.BackgroundColor3 = Color3.fromRGB(70, 70, 70)
            btn.TextColor3 = Color3.new(1, 1, 1)
            btn.TextXAlignment = Enum.TextXAlignment.Left
            btn.MouseButton1Click:Connect(function()
                Input.Text = player.Name
                Suggestions.Visible = false
            end)
            btn.Parent = Suggestions
        end
    end
    Suggestions.CanvasSize = UDim2.new(0, 0, 0, #Suggestions:GetChildren() * 22)
    Suggestions.Visible = #Suggestions:GetChildren() > 0
end

Input.Focused:Connect(UpdateSuggestions)
Input.Changed:Connect(UpdateSuggestions)

-- Fling System (Working)
local Flinging = false
local function FlingTarget(target)
    while Flinging and wait(0.02) do
        if target.Character and LP.Character then
            local tHRP = target.Character:FindFirstChild("HumanoidRootPart")
            local myHRP = LP.Character:FindFirstChild("HumanoidRootPart")
            
            if tHRP and myHRP then
                -- Prediction pattern
                local offsets = {
                    Vector3.new(8, 12, 8),
                    Vector3.new(-8, 12, -8),
                    Vector3.new(12, 8, -12),
                    Vector3.new(-12, 8, 12)
                }
                
                for _, offset in ipairs(offsets) do
                    if not Flinging then break end
                    myHRP.CFrame = tHRP.CFrame + offset
                    myHRP.Velocity = (tHRP.Position - myHRP.Position).Unit * 8500 + Vector3.new(0, 6500, 0)
                    myHRP.CanCollide = false
                    tHRP.CanCollide = false
                    wait(0.03)
                end
            end
        end
    end
end

-- Camera System
local Tracking = false
local CameraDistance = 15

local function TrackTarget(target)
    while Tracking and wait() do
        if target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
            Camera.CameraType = Enum.CameraType.Scriptable
            Camera.CFrame = CFrame.new(
                target.Character.HumanoidRootPart.Position + Vector3.new(0, 5, CameraDistance),
                target.Character.HumanoidRootPart.Position
            )
        end
    end
    Camera.CameraType = Enum.CameraType.Custom
end

-- Button Handlers
Toggle.MouseButton1Click:Connect(function()
    Flinging = not Flinging
    if Flinging then
        local target = Players:FindFirstChild(Input.Text)
        if target then
            coroutine.wrap(FlingTarget)(target)
        end
    end
end)

CamButton.MouseButton1Click:Connect(function()
    Tracking = not Tracking
    if Tracking then
        local target = Players:FindFirstChild(Input.Text)
        if target then
            coroutine.wrap(TrackTarget)(target)
        end
    else
        Camera.CameraType = Enum.CameraType.Custom
    end
end)

-- Zoom Control
UIS.InputChanged:Connect(function(input)
    if Tracking and input.UserInputType == Enum.UserInputType.MouseWheel then
        CameraDistance = math.clamp(CameraDistance - input.Position.Z * 3, 8, 20)
    end
end)
