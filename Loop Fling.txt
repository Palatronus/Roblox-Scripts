-- Original alignment function preserved
local function align(Part0, Part1)
    local att1 = Instance.new("Attachment")
    att1.Position = Vector3.new(0, 0, 0)
    local att0 = att1:Clone()

    local ap = Instance.new("AlignPosition", att0)
    ap.ApplyAtCenterOfMass = true
    ap.MaxForce = 9e9
    ap.MaxVelocity = 9e9
    ap.ReactionForceEnabled = false
    ap.Responsiveness = 200
    ap.RigidityEnabled = false
    ap.Attachment1 = att1
    ap.Attachment0 = att0

    att1.Parent = Part1
    att0.Parent = Part0
end

-- Drag functionality
local dragging
local dragInput
local dragStart
local startPos

local function update(input)
    local delta = input.Position - dragStart
    frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

title.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

title.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        update(input)
    end
end)

-- Fling core logic
local flingActive = false
local connections = {}
local anchorParts = {}

local function getTargetPlayer()
    for _, player in ipairs(Players:GetPlayers()) do
        if player.Name:lower() == targetBox.Text:lower() then
            return player
        end
    end
end

local function maintainFling(targetChar)
    if not LocalPlayer.Character then return end
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    -- Store original state
    local originalCFrame = hrp.CFrame
    local originalRot = hrp.RotVelocity
    local originalVelocity = hrp.Velocity

    -- Continuous fling loop
    while flingActive and targetChar and targetChar.Parent do
        if not targetChar:FindFirstChild("HumanoidRootPart") then
            task.wait(0.1)
            continue
        end

        -- Teleport to target
        hrp.CFrame = targetChar.HumanoidRootPart.CFrame

        -- Create fling anchor
        if not anchorParts[targetChar] then
            local anchor = Instance.new("Part")
            anchor.Name = "FlingAnchor"
            anchor.Anchored = true
            anchor.Transparency = 1
            anchor.CanCollide = false
            anchor.Size = Vector3.new(2, 2, 1)
            anchor.CFrame = hrp.CFrame
            anchor.Parent = workspace
            anchorParts[targetChar] = anchor

            -- Align parts
            for _, part in ipairs(LocalPlayer.Character:GetDescendants()) do
                if part:IsA("BasePart") then
                    if part.CanCollide then
                        align(part, anchor)
                    else
                        part.Velocity = Vector3.new(0, -1000, 0)
                    end
                end
            end
        end

        -- Maintain velocity
        RunService.Heartbeat:Wait()
        for _, part in ipairs(LocalPlayer.Character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Velocity = Vector3.new(0, -25.05, 0)
                part.RotVelocity = Vector3.new(9999, 9999, 9999)
            end
        end

        -- Reset position periodically
        task.wait(0.1)
        hrp.CFrame = originalCFrame
        hrp.Velocity = originalVelocity
        hrp.RotVelocity = originalRot
    end

    -- Cleanup
    if anchorParts[targetChar] then
        anchorParts[targetChar]:Destroy()
        anchorParts[targetChar] = nil
    end
end

toggleBtn.MouseButton1Click:Connect(function()
    flingActive = not flingActive
    toggleBtn.Text = flingActive and "Stop Flinging" or "Start Flinging"
    
    if flingActive then
        local targetPlayer = getTargetPlayer()
        if not targetPlayer then return end
        
        -- Character tracking
        local function trackCharacter(char)
            if char then
                maintainFling(char)
            end
        end
        
        trackCharacter(targetPlayer.Character)
        connections[targetPlayer] = targetPlayer.CharacterAdded:Connect(trackCharacter)
    else
        -- Disconnect all connections
        for player, conn in pairs(connections) do
            conn:Disconnect()
            connections[player] = nil
        end
        -- Clean anchors
        for _, anchor in pairs(anchorParts) do
            anchor:Destroy()
        end
        table.clear(anchorParts)
    end
end)

-- Auto-reinject on character respawn
LocalPlayer.CharacterAdded:Connect(function()
    if flingActive then
        task.wait(1)
        toggleBtn.Text = "Stop Flinging"
    end
end)
