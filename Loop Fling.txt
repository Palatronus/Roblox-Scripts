local function FlingTarget(target)
    while Flinging and task.wait(0.01) do -- Faster update rate
        if target.Character and LP.Character then
            local tHRP = target.Character:FindFirstChild("HumanoidRootPart")
            local myHRP = LP.Character:FindFirstChild("HumanoidRootPart")
            
            if tHRP and myHRP then
                -- Chaos Pattern Matrix
                local chaosOffset = CFrame.new(
                    math.random(-5,5) * math.pi,  -- Spiral X
                    math.random(2,8),              -- Forced vertical
                    math.random(-5,5) * math.pi    -- Spiral Z
                )

                -- Extreme Velocity Surge
                myHRP.CFrame = tHRP.CFrame * chaosOffset
                myHRP.Velocity = (tHRP.Position - myHRP.Position).Unit * 12000 
                                + Vector3.new(0, 6000, 0)
                
                -- Target Disorientation
                tHRP.Velocity = Vector3.new(
                    math.random(-5000,5000),
                    math.random(2000,5000),
                    math.random(-5000,5000)
                )
                tHRP.RotVelocity = Vector3.new(
                    math.random(-50,50),
                    math.random(-50,50),
                    math.random(-50,50)
                )
                
                -- Collision Enforcement
                myHRP.CanCollide = false
                tHRP.CanCollide = false
            end
        end
    end
end
