-- Enhanced Impossible-to-Dodge Fling System
local Flinging = false
local function FlingTarget(target)
    while Flinging and task.wait(0.01) do  -- Faster update cycle
        if target.Character and LP.Character then
            local tHRP = target.Character:FindFirstChild("HumanoidRootPart")
            local myHRP = LP.Character:FindFirstChild("HumanoidRootPart")
            
            if tHRP and myHRP then
                -- Chaotic Orbital Pattern
                local angle = tick() * 20  -- Time-based rotation
                local orbitOffset = Vector3.new(
                    math.cos(angle) * 5,   -- X axis orbit (spiral effect)
                    math.random(1, 4),     -- Vertical chaos
                    math.sin(angle) * 5    -- Z axis orbit (spiral effect)
                )

                -- Extreme Velocity Forces
                myHRP.CFrame = tHRP.CFrame + orbitOffset
                myHRP.Velocity = (tHRP.Position - myHRP.Position).Unit * 15000 
                                + Vector3.new(0, 8000, 0)
                
                -- Target Disruption System
                myHRP.RotVelocity = Vector3.new(
                    math.random(-500,500),
                    math.random(-500,500),
                    math.random(-500,500)
                )

                -- Aggressive Collision Bypass
                myHRP.CanCollide = false
                tHRP.CanCollide = false
                
                -- Anti-Recovery Measures
                tHRP.Velocity = Vector3.new(
                    math.random(-10000,10000),
                    math.random(5000,10000),
                    math.random(-10000,10000)
                )
            end
        end
    end
end
