-- Enhanced Fling System
local Flinging = false
local function FlingTarget(target)
    while Flinging do
        if target and target.Character and LP.Character then
            local tHRP = target.Character:FindFirstChild("HumanoidRootPart")
            local myHRP = LP.Character:FindFirstChild("HumanoidRootPart")
            
            if tHRP and myHRP then
                -- Nameless Admin-style pattern
                local pattern = {
                    Vector3.new(8, 12, 8),
                    Vector3.new(-8, 12, -8),
                    Vector3.new(12, 8, -12),
                    Vector3.new(-12, 8, 12)
                }
                
                -- Cycle through positions with collision bypass
                for _, offset in ipairs(pattern) do
                    if not Flinging then break end
                    
                    -- Phase through target
                    myHRP.CanCollide = false
                    tHRP.CanCollide = false
                    
                    -- Position and velocity
                    myHRP.CFrame = tHRP.CFrame + offset
                    myHRP.Velocity = (tHRP.Position - myHRP.Position).Unit * 10000 + Vector3.new(0, 8000, 0)
                    
                    wait(0.03) -- Original Nameless timing
                end
            end
        end
        wait() -- Yield for Lua 5.1 compatibility
    end
end
